const INTERVAL = 100;

class Timer extends React.Component {
constructor(props) {
super(props);
this.state = {value: 0};
}

increment(){
this.setState({value: this.state.value + 1});
}

componentDidMount() {
this.timerID = setInterval(() => this.increment(), 1000/INTERVAL);
}

componentWillUnmount() {
clearInterval(this.timerID);
}
